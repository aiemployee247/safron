    <script>
      const ROLES = { foreman: "coordinator", radar: "research", quill: "writing", wrench: "engineering", ledger: "numbers" };
      const fmt = (ts) => (ts ? ts.replace("T", " ").slice(5, 16) : "—");
      const money = (n) => "$" + (Number(n) || 0).toFixed(Number(n) >= 1 ? 2 : 4);
      const num = (n) => (Number(n) || 0).toLocaleString();
      const esc = (s) => String(s).replace(/[&<>"]/g, (c) => ({ "&": "&amp;", "<": "&lt;", ">": "&gt;", '"': "&quot;" }[c]));

      function tickClock() {
        document.getElementById("clock").textContent = new Date().toISOString().slice(11, 19);
      }
      tickClock();
      setInterval(tickClock, 1000);

      document.getElementById("tabs").addEventListener("click", (e) => {
        const btn = e.target.closest(".tab"); if (!btn) return;
        document.querySelectorAll(".tab").forEach((t) => t.classList.toggle("active", t === btn));
        document.querySelectorAll(".view").forEach((v) => v.classList.toggle("active", v.id === "view-" + btn.dataset.view));
        if (btn.dataset.view === "host") loadHost();
        if (btn.dataset.view === "kanban") loadKanban();
        if (btn.dataset.view === "docs") loadDocs();
      });

      // ---- Host ----
      function fmtUptime(sec) {
        const d = Math.floor(sec / 86400), h = Math.floor((sec % 86400) / 3600), m = Math.floor((sec % 3600) / 60);
        return (d ? d + "d " : "") + h + "h " + m + "m";
      }
      function fmtBytes(n) {
        const gb = n / 1024 / 1024 / 1024;
        return gb >= 1 ? gb.toFixed(1) + " GB" : (n / 1024 / 1024).toFixed(0) + " MB";
      }
      async function loadHost() {
        try {
          const h = await fetch("/api/host").then((r) => r.json());
          document.getElementById("host-platform").textContent = h.platform === "darwin" ? "macOS" : h.platform;
          document.getElementById("host-uptime").textContent = fmtUptime(h.uptimeSec);
          document.getElementById("host-load").textContent = (h.loadAvg[0] || 0).toFixed(2);
          document.getElementById("host-mem").textContent = fmtBytes(h.memUsed) + " / " + fmtBytes(h.memTotal);
          document.getElementById("host-jobs").innerHTML = h.jobs && h.jobs.length
            ? h.jobs.map((j) => '<div class="rrow"><span class="a">' + esc(j.status || j.active || "") + '</span><span class="s">' + esc(j.label) + "</span></div>").join("")
            : '<div class="empty">no jobs found for this crew</div>';
        } catch { /* leave last-known values on screen */ }
      }

      // ---- Kanban ----
      const KCOLS = [["todo", "To do"], ["doing", "In progress"], ["done", "Done"]];
      async function loadKanban() {
        const board = await fetch("/api/kanban").then((r) => r.json());
        document.getElementById("kanban-board").innerHTML = KCOLS.map(([key, label]) => {
          const cards = (board[key] || []).map((c) =>
            '<div class="kcard" data-id="' + c.id + '"><span class="txt">' + esc(c.text) + '</span>' +
            (key !== "done" ? '<button data-move="' + (key === "todo" ? "doing" : "done") + '" data-id="' + c.id + '" title="Move forward">→</button>' : '') +
            '<button data-del="' + c.id + '" title="Delete">×</button></div>'
          ).join("") || '<div class="empty">empty</div>';
          return '<div class="kcol"><h3>' + label + '</h3><div data-col="' + key + '">' + cards + '</div>' +
            (key === "todo" ? '<div class="kadd"><input id="kadd-input" placeholder="Add a card…" /><button id="kadd-btn">Add</button></div>' : '') +
            '</div>';
        }).join("");

        document.getElementById("kanban-board").querySelectorAll("[data-move]").forEach((b) =>
          b.addEventListener("click", async () => {
            await fetch("/api/kanban", { method: "POST", headers: { "Content-Type": "application/json" },
              body: JSON.stringify({ action: "move", id: b.dataset.id, to: b.dataset.move }) });
            loadKanban();
          }));
        document.getElementById("kanban-board").querySelectorAll("[data-del]").forEach((b) =>
          b.addEventListener("click", async () => {
            await fetch("/api/kanban", { method: "POST", headers: { "Content-Type": "application/json" },
              body: JSON.stringify({ action: "delete", id: b.dataset.del }) });
            loadKanban();
          }));
        const addBtn = document.getElementById("kadd-btn");
        if (addBtn) addBtn.addEventListener("click", async () => {
          const input = document.getElementById("kadd-input");
          const text = input.value.trim();
          if (!text) return;
          await fetch("/api/kanban", { method: "POST", headers: { "Content-Type": "application/json" },
            body: JSON.stringify({ action: "add", column: "todo", text }) });
          input.value = "";
          loadKanban();
        });
      }

      // ---- Docs ----
      let currentDoc = null;
      async function loadDocs() {
        const files = await fetch("/api/content").then((r) => r.json());
        document.getElementById("doc-list").innerHTML = files.length
          ? files.map((f) => '<div class="doc-item' + (f.file === currentDoc ? " active" : "") + '" data-file="' + esc(f.file) + '">' + esc(f.file) + "</div>").join("")
          : '<div class="empty">no notes yet</div>';
        document.getElementById("doc-list").querySelectorAll("[data-file]").forEach((el) =>
          el.addEventListener("click", () => openDoc(el.dataset.file)));
      }
      async function openDoc(file) {
        const r = await fetch("/api/content?file=" + encodeURIComponent(file)).then((r) => r.json());
        currentDoc = file;
        document.getElementById("doc-current").textContent = file;
        const ta = document.getElementById("doc-body");
        ta.value = r.body || "";
        ta.disabled = false;
        document.getElementById("doc-save-btn").disabled = false;
        document.getElementById("doc-delete-btn").disabled = false;
        loadDocs();
      }
      document.getElementById("doc-new-btn").addEventListener("click", async () => {
        const nameInput = document.getElementById("doc-new-name");
        let name = nameInput.value.trim();
        if (!name) return;
        if (!/\.(md|txt)$/.test(name)) name += ".md";
        await fetch("/api/content?file=" + encodeURIComponent(name), { method: "POST",
          headers: { "Content-Type": "application/json" }, body: JSON.stringify({ body: "" }) });
        nameInput.value = "";
        await openDoc(name);
      });
      document.getElementById("doc-save-btn").addEventListener("click", async () => {
        if (!currentDoc) return;
        await fetch("/api/content?file=" + encodeURIComponent(currentDoc), { method: "POST",
          headers: { "Content-Type": "application/json" }, body: JSON.stringify({ body: document.getElementById("doc-body").value }) });
      });
      document.getElementById("doc-delete-btn").addEventListener("click", async () => {
        if (!currentDoc || !confirm("Delete " + currentDoc + "?")) return;
        await fetch("/api/content?file=" + encodeURIComponent(currentDoc), { method: "DELETE" });
        currentDoc = null;
        document.getElementById("doc-current").textContent = "Select a note";
        const ta = document.getElementById("doc-body");
        ta.value = ""; ta.disabled = true;
        document.getElementById("doc-save-btn").disabled = true;
        document.getElementById("doc-delete-btn").disabled = true;
        loadDocs();
      });

      function fleetRow(a) {
        const dot = '<span class="dot ' + (a.status === "active" ? "active" : "") + '"></span>';
        return '<div class="fleet-row">' + dot +

          '<span class="fleet-role">' + (ROLES[a.agent] || "") + '</span>' +
          '<span class="fleet-meta"><span>turns <b>' + num(a.turns) + '</b></span><span>cost <b>' + money(a.cost) + '</b></span><span>' + fmt(a.lastActivity) + '</span></span>' +
          '</div>';
      }

        const dot = '<span class="dot ' + (a.status === "active" ? "active" : "") + '"></span>';
        return '<div class="panel agent-card b-4 rise" style="animation-delay:' + (i * 40) + 'ms">' +

            '<div class="meta-tile"><div class="l">Tok out</div><div class="v">' + num(a.tokOut) + '</div></div>' +
          '</div>' +
          '<div class="stat-sub">last active ' + fmt(a.lastActivity) + '</div>' +
        '</div>';
      }

      async function tick() {
        try {
          const [stats, agents, events, tasks] = await Promise.all([
            fetch("/api/stats").then((r) => r.json()),
            fetch("/api/agents").then((r) => r.json()),
            fetch("/api/events?limit=40").then((r) => r.json()),

          document.getElementById("kpi-active").textContent = stats.agentsActive + " / " + stats.agentsTotal;

          document.getElementById("kpi-turns-total").textContent = num(stats.turnsTotal) + " all-time";
          document.getElementById("kpi-cost").textContent = money(stats.costToday);
          document.getElementById("kpi-cost-total").textContent = money(stats.costTotal) + " all-time";
          document.getElementById("kpi-errors").textContent = num(stats.errorsToday);

          const feedHtml = events.length
            ? events.map((e) => '<div class="rrow"><span class="t">' + fmt(e.ts) + '</span><span class="a">' + e.agent +
                '</span><span class="k">' + e.kind + '</span><span class="s">' + esc(e.summary) + "</span></div>").join("")
            : '<div class="empty">no events yet</div>';
          document.getElementById("feed-mini").innerHTML = events.length
            ? events.slice(0, 8).map((e) => '<div class="rrow"><span class="t">' + fmt(e.ts) + '</span><span class="s">' + esc(e.summary) + "</span></div>").join("")
            : '<div class="empty">waiting for activity…</div>';

          document.getElementById("fleet-list").innerHTML = agents.map(fleetRow).join("");
          document.getElementById("agent-cards").innerHTML = agents.map(agentCard).join("");

          // focus ring: agent with the largest share of today's turns
          const top = agents.reduce((best, a) => (a.turns > (best?.turns || 0) ? a : best), null);
          const totalTurns = agents.reduce((s, a) => s + (Number(a.turns) || 0), 0);
          const pct = top && totalTurns ? Math.round((top.turns / totalTurns) * 100) : 0;
          document.getElementById("focus-pct").textContent = pct + "%";
          document.getElementById("focus-agent").textContent = (top ? top.agent : "—") + " of turns";
          const circumference = 2 * Math.PI * 42;
          document.getElementById("focus-arc").setAttribute("stroke-dasharray", (circumference * pct / 100).toFixed(1) + " " + circumference.toFixed(1));

          document.getElementById("tasks-open").innerHTML = tasks.open && tasks.open.length
            ? tasks.open.map((t) => '<div class="rrow"><span class="a">' + t.agent + '</span><span class="s">' + esc(t.summary) + "</span></div>").join("")
            : '<div class="empty">nothing running</div>';
          document.getElementById("tasks-done").innerHTML = tasks.done && tasks.done.length
            ? tasks.done.map((t) => '<div class="rrow"><span class="a">' + t.agent + '</span><span class="s">' + esc(t.summary) +
                '</span><span class="t" style="margin-left:auto">' + fmt(t.ts) + "</span></div>").join("")
            : '<div class="empty">nothing yet</div>';

          document.getElementById("mon-model").textContent = stats.model || "—";
          document.getElementById("mon-in").textContent = num(stats.tokInToday);
          document.getElementById("mon-out").textContent = num(stats.tokOutToday);
          document.getElementById("mon-total").textContent = money(stats.costTotal);
          const maxCost = Math.max(0.000001, ...agents.map((a) => Number(a.cost) || 0));
          document.getElementById("mon-table").innerHTML = agents.map((a) =>
            '<tr><td class="agent">' + a.agent + '</td><td class="num">' + num(a.turns) + '</td><td class="num">' +

            '<div class="bar"><span style="width:' + Math.round((Number(a.cost) || 0) / maxCost * 100) + '%"></span></div></td></tr>').join("");

          document.getElementById("status").textContent = "live · updated " + new Date().toLocaleTimeString();
        } catch (err) {
          document.getElementById("status").textContent = "reconnecting…";
        }
      }
      tick();
      setInterval(tick, 4000);
