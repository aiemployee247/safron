        const dot = '<span class="dot ' + (a.status === "active" ? "active" : "") + '"></span>';
        return '<div class="panel agent-card b-4 rise" style="animation-delay:' + (i * 40) + 'ms; cursor:pointer;" data-open-agent="' + a.agent + '">' +
          '<div class="name-row">' + dot + a.agent + '</div>' +
          '<div class="role">' + (ROLES[a.agent] || "") + '</div>' +
          '<div class="meta-grid">' +
            '<div class="meta-tile"><div class="l">Turns</div><div class="v">' + num(a.turns) + '</div></div>' +
            '<div class="meta-tile"><div class="l">Cost</div><div class="v">' + money(a.cost) + '</div></div>' +
            '<div class="meta-tile"><div class="l">Tok in</div><div class="v">' + num(a.tokIn) + '</div></div>' +
            '<div class="meta-tile"><div class="l">Tok out</div><div class="v">' + num(a.tokOut) + '</div></div>' +
          '</div>' +
          '<div class="stat-sub">last active ' + fmt(a.lastActivity) + ' · click for detail</div>' +
        '</div>';
      }

      async function tick() {
        try {
          const [stats, agents, events, tasks] = await Promise.all([
            fetch("/api/stats").then((r) => r.json()),
            fetch("/api/agents").then((r) => r.json()),
            fetch("/api/events?limit=40").then((r) => r.json()),
            fetch("/api/tasks").then((r) => r.json()),
          ]);

          document.getElementById("kpi-active").textContent = stats.agentsActive + " / " + stats.agentsTotal;
          document.getElementById("hero-active-pill").textContent = stats.agentsActive + " / " + stats.agentsTotal + " active";
          document.getElementById("kpi-turns").textContent = num(stats.turnsToday);
          document.getElementById("kpi-turns-total").textContent = num(stats.turnsTotal) + " all-time";
          document.getElementById("kpi-cost").textContent = money(stats.costToday);
          document.getElementById("kpi-cost-total").textContent = money(stats.costTotal) + " all-time";
          document.getElementById("kpi-errors").textContent = num(stats.errorsToday);

          const feedHtml = events.length
            ? events.map((e) => '<div class="rrow"><span class="t">' + fmt(e.ts) + '</span><span class="a">' + e.agent +
                '</span><span class="k">' + e.kind + '</span><span class="s">' + esc(e.summary) + "</span></div>").join("")
            : '<div class="empty">no events yet</div>';
          document.getElementById("feed-mini").innerHTML = events.length
            ? events.slice(0, 8).map((e) => '<div class="rrow"><span class="t">' + fmt(e.ts) + '</span><span class="s">' + esc(e.summary) + "</span></div>").join("")
            : '<div class="empty">waiting for activity…</div>';

          document.getElementById("fleet-list").innerHTML = agents.map(fleetRow).join("");
          document.getElementById("agent-cards").innerHTML = agents.map(agentCard).join("");
          document.getElementById("agent-cards").querySelectorAll("[data-open-agent]").forEach((el) =>
            el.addEventListener("click", () => openDrawer(el.dataset.openAgent)));

          // focus ring: agent with the largest share of today's turns
          const top = agents.reduce((best, a) => (a.turns > (best?.turns || 0) ? a : best), null);
          const totalTurns = agents.reduce((s, a) => s + (Number(a.turns) || 0), 0);
          const pct = top && totalTurns ? Math.round((top.turns / totalTurns) * 100) : 0;
          document.getElementById("focus-pct").textContent = pct + "%";
          document.getElementById("focus-agent").textContent = (top ? top.agent : "—") + " of turns";
          const circumference = 2 * Math.PI * 42;
          document.getElementById("focus-arc").setAttribute("stroke-dasharray", (circumference * pct / 100).toFixed(1) + " " + circumference.toFixed(1));

          document.getElementById("tasks-open").innerHTML = tasks.open && tasks.open.length
            ? tasks.open.map((t) => '<div class="rrow"><span class="a">' + t.agent + '</span><span class="s">' + esc(t.summary) + "</span></div>").join("")
            : '<div class="empty">nothing running</div>';
          document.getElementById("tasks-done").innerHTML = tasks.done && tasks.done.length
            ? tasks.done.map((t) => '<div class="rrow"><span class="a">' + t.agent + '</span><span class="s">' + esc(t.summary) +
                '</span><span class="t" style="margin-left:auto">' + fmt(t.ts) + "</span></div>").join("")
            : '<div class="empty">nothing yet</div>';

          document.getElementById("mon-model").textContent = stats.model || "—";
          document.getElementById("mon-in").textContent = num(stats.tokInToday);
          document.getElementById("mon-out").textContent = num(stats.tokOutToday);
          document.getElementById("mon-total").textContent = money(stats.costTotal);
          const maxCost = Math.max(0.000001, ...agents.map((a) => Number(a.cost) || 0));
          document.getElementById("mon-table").innerHTML = agents.map((a) =>
            '<tr><td class="agent">' + a.agent + '</td><td class="num">' + num(a.turns) + '</td><td class="num">' +
            num(a.tokIn) + '</td><td class="num">' + num(a.tokOut) + '</td><td class="num">' + money(a.cost) +
            '<div class="bar"><span style="width:' + Math.round((Number(a.cost) || 0) / maxCost * 100) + '%"></span></div></td></tr>').join("");

          document.getElementById("status").textContent = "live · updated " + new Date().toLocaleTimeString();
        } catch (err) {
          document.getElementById("status").textContent = "reconnecting…";
        }
      }
      tick();
      setInterval(tick, 4000);
    </script>

    <!-- Office — a 3D shop floor, one building per agent. This is the only
         external dependency in the whole dashboard (Three.js, via CDN + an
         import map). Lazy: nothing here loads or runs until the Office tab
         is opened for the first time. -->
    <script type="module">
      const AGENT_COLORS = {
        foreman: 0xeac266, radar: 0x4bc6fa, quill: 0xad8cff, wrench: 0xff7c68, ledger: 0x00c89c,
      };
      const LAYOUT_ORDER = ["radar", "quill", "wrench", "ledger"]; // ring around the Foreman's tower

      let officeStarted = false;
      let scene, camera, renderer, controls, raycaster, mouse, animId;
      const buildings = {}; // agent key -> { mesh, glow, targetEmissive }

      async function initOffice() {
        if (officeStarted) return;
        officeStarted = true;
        try {
          const THREE = await import("three");
          const { OrbitControls } = await import("three/addons/controls/OrbitControls.js");
          buildScene(THREE, OrbitControls);
        } catch (err) {
          const wrap = document.getElementById("office-canvas-wrap");
          wrap.innerHTML = '<div class="office-loading"><div class="t1">Could not load the 3D view</div>' +
            '<div class="t2">' + (err && err.message ? err.message : "WebGL may be unavailable in this browser.") + '</div></div>';
        }
      }
      window.initOffice = initOffice;

      function buildScene(THREE, OrbitControls) {
        const wrap = document.getElementById("office-canvas-wrap");
        const width = wrap.clientWidth, height = wrap.clientHeight;

        scene = new THREE.Scene();
        scene.background = new THREE.Color(0x04060c);
        scene.fog = new THREE.Fog(0x04060c, 14, 34);

        camera = new THREE.PerspectiveCamera(45, width / height, 0.1, 100);
        camera.position.set(9, 8, 11);

        renderer = new THREE.WebGLRenderer({ antialias: true, alpha: false });
        renderer.setPixelRatio(Math.min(window.devicePixelRatio, 2));
        renderer.setSize(width, height);
        renderer.shadowMap.enabled = true;
        wrap.insertBefore(renderer.domElement, wrap.firstChild);

        controls = new OrbitControls(camera, renderer.domElement);
        controls.enableDamping = true;
        controls.dampingFactor = 0.08;
        controls.minDistance = 5;
        controls.maxDistance = 26;
        controls.maxPolarAngle = Math.PI / 2.05;
        controls.target.set(0, 1, 0);

        scene.add(new THREE.AmbientLight(0x445566, 1.1));
        const moon = new THREE.DirectionalLight(0x8fa8ff, 0.5);
        moon.position.set(-8, 12, -6);
        scene.add(moon);

        // ground
        const groundGeo = new THREE.CircleGeometry(16, 48);
        const groundMat = new THREE.MeshStandardMaterial({ color: 0x0a0e18, roughness: 0.95 });
        const ground = new THREE.Mesh(groundGeo, groundMat);
        ground.rotation.x = -Math.PI / 2;
        ground.receiveShadow = true;
        scene.add(ground);
        const grid = new THREE.GridHelper(32, 32, 0x2a3244, 0x181e2c);
        grid.position.y = 0.01;
        scene.add(grid);

        // Foreman's HQ tower, centre
        addBuilding(THREE, "foreman", 0, 0, 1.6, 5.2);
        // four specialists in a ring
        const radius = 5.2;
        LAYOUT_ORDER.forEach((key, i) => {
          const angle = (i / LAYOUT_ORDER.length) * Math.PI * 2;
          addBuilding(THREE, key, Math.cos(angle) * radius, Math.sin(angle) * radius, 1.1, 3.2);
        });

        raycaster = new THREE.Raycaster();
        mouse = new THREE.Vector2();
        renderer.domElement.addEventListener("click", (e) => onClick(THREE, e));

        window.addEventListener("resize", onResize);

        document.getElementById("office-loading").style.display = "none";
        renderLegend();
        animate(THREE);
        pollOfficeData();
        setInterval(pollOfficeData, 4000);
      }

      function addBuilding(THREE, key, x, z, footprint, height) {
        const color = AGENT_COLORS[key] || 0x888888;
        const geo = new THREE.BoxGeometry(footprint, height, footprint);
        const mat = new THREE.MeshStandardMaterial({
          color: 0x11151f, emissive: new THREE.Color(color), emissiveIntensity: 0.12,
          roughness: 0.6, metalness: 0.15,
        });
        const mesh = new THREE.Mesh(geo, mat);
        mesh.position.set(x, height / 2, z);
        mesh.castShadow = true;
        mesh.userData.agentKey = key;
        scene.add(mesh);

        const glow = new THREE.PointLight(color, 0, 6);
        glow.position.set(x, height + 0.6, z);
        scene.add(glow);

        buildings[key] = { mesh, mat, glow, targetEmissive: 0.12, targetGlow: 0 };
      }

      function onResize() {
        const wrap = document.getElementById("office-canvas-wrap");
        if (!wrap || !renderer) return;
        const width = wrap.clientWidth, height = wrap.clientHeight;
        camera.aspect = width / height;
        camera.updateProjectionMatrix();
        renderer.setSize(width, height);
      }

      function onClick(THREE, e) {
        const wrap = document.getElementById("office-canvas-wrap");
        const rect = wrap.getBoundingClientRect();
        mouse.x = ((e.clientX - rect.left) / rect.width) * 2 - 1;
        mouse.y = -((e.clientY - rect.top) / rect.height) * 2 + 1;
        raycaster.setFromCamera(mouse, camera);
        const meshes = Object.values(buildings).map((b) => b.mesh);
        const hit = raycaster.intersectObjects(meshes)[0];
        if (hit && hit.object.userData.agentKey && window.openDrawer) {
          window.openDrawer(hit.object.userData.agentKey);
        }
      }

      function animate(THREE) {
        animId = requestAnimationFrame(() => animate(THREE));
        controls.update();
        for (const key in buildings) {
          const b = buildings[key];
          b.mat.emissiveIntensity += (b.targetEmissive - b.mat.emissiveIntensity) * 0.08;
          b.glow.intensity += (b.targetGlow - b.glow.intensity) * 0.08;
        }
        renderer.render(scene, camera);
      }

      async function pollOfficeData() {
        try {
          const [agents, stats] = await Promise.all([
            fetch("/api/agents").then((r) => r.json()),
            fetch("/api/stats").then((r) => r.json()),
          ]);
          let lit = 0;
          agents.forEach((a) => {
            const b = buildings[a.agent];
            if (!b) return;
            const active = a.status === "active";
            if (active) lit++;
            b.targetEmissive = active ? 0.85 : 0.12;
            b.targetGlow = active ? 2.4 : 0;
          });
          document.getElementById("office-lights").textContent = lit;
          document.getElementById("office-turns").textContent = stats.turnsToday ?? 0;
        } catch { /* keep last-known glow state */ }
      }

      function renderLegend() {
        const roster = Object.keys(AGENT_COLORS);
        document.getElementById("office-legend").innerHTML = roster.map((key) => {
          const hex = "#" + AGENT_COLORS[key].toString(16).padStart(6, "0");
          return '<div class="office-legend-item"><span class="office-legend-dot" style="background:' + hex + ';"></span>' + key + '</div>';
        }).join("");
      }
    </script>
